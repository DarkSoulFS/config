035/**
036 * This is the object mapper. It handles conversion between configuration nodes and
037 * fields annotated with {@link Setting} in objects.
038 *
039 * Values in the node not used by the mapped object will be preserved.
040 *
041 * @param <T> The type to work with
042 */
043public class ObjectMapper<T> {
044    private final TypeToken<T> type;
045    private final Class<? super T> clazz;
046    @Nullable
047    private final Invokable<T, T> constructor;
048    private final Map<String, FieldData> cachedFields = new LinkedHashMap<>();
049
050
051    /**
052     * Create a new object mapper that can work with objects of the given class using the
053     * {@link DefaultObjectMapperFactory}.
054     *
055     * @param clazz The type of object
056     * @param <T> The type
057     * @return An appropriate object mapper instance. May be shared with other users.
058     * @throws ObjectMappingException If invalid annotated fields are presented
059     */
060    public static <T> ObjectMapper<T> forClass(@NonNull Class<T> clazz) throws ObjectMappingException {
061        return DefaultObjectMapperFactory.getInstance().getMapper(clazz);
062    }
063
064    /**
065     * Create a new object mapper that can work with objects of the given type using the
066     * {@link DefaultObjectMapperFactory}.
067     *
068     * @param type The type of object
069     * @param <T> The type
070     * @return An appropriate object mapper instance. May be shared with other users.
071     * @throws ObjectMappingException If invalid annotated fields are presented
072     */
073    public static <T> ObjectMapper<T> forType(@NonNull TypeToken<T> type) throws ObjectMappingException {
074        return DefaultObjectMapperFactory.getInstance().getMapper(type);
075    }
076
077    /**
078     * Creates a new object mapper bound to the given object.
079     *
080     * <strong>CAUTION</strong> Generic type information will be lost when creating a mapper. Provide a TypeToken to avoid this
081     *
082     * @param obj The object
083     * @param <T> The object type
084     * @return An appropriate object mapper instance.
085     * @throws ObjectMappingException when an object is provided that is not suitable for object mapping.
086     *                              Reasons may include but are not limited to:
087     *                              <ul>
088     *                                  <li>Not annotated with {@link ConfigSerializable} annotation</li>
089     *                                  <li>Invalid field types</li>
090     *                              </ul>
091     */
092    @SuppressWarnings("unchecked")
093    public static <T> ObjectMapper<T>.BoundInstance forObject(@NonNull T obj) throws ObjectMappingException {
094        return forClass((Class<T>) requireNonNull(obj).getClass()).bind(obj);
095    }
096
097    /**
098     * Creates a new object mapper bound to the given object.
099     *
100     * @param type The generic type of the object
101     * @param obj The object
102     * @param <T> The object type
103     * @return An appropriate object mapper instance.
104     * @throws ObjectMappingException when an object is provided that is not suitable for object mapping.
105     *                              Reasons may include but are not limited to:
106     *                              <ul>
107     *                                  <li>Not annotated with {@link ConfigSerializable} annotation</li>
108     *                                  <li>Invalid field types</li>
109     *                                  <li>Specified type is an interface</li>
110     *                              </ul>
111     */
112    @SuppressWarnings("unchecked")
113    public static <T> ObjectMapper<T>.BoundInstance forObject(TypeToken<T> type, @NonNull T obj) throws ObjectMappingException {
114        return forType(requireNonNull(type)).bind(obj);
115    }

117    /**
118     * Holder for field-specific information
119     */
120    protected static class FieldData {
121        private final Field field;
122        private final TypeToken<?> fieldType;
123        private final String comment;
124
125        public FieldData(Field field, String comment, TypeToken<?> resolvedFieldType) {
126            this.field = field;
127            this.comment = comment;
128            this.fieldType = resolvedFieldType;
129        }
130
131        public <Node extends ScopedConfigurationNode<Node>> void deserializeFrom(Object instance, Node node) throws ObjectMappingException {
132            TypeSerializer<?> serial = node.getOptions().getSerializers().get(this.fieldType);
133            if (serial == null) {
134                throw new ObjectMappingException("No TypeSerializer found for field " + field.getName() + " of type "
135                        + this.fieldType);
136            }
137            Object newVal = node.isVirtual() ? null : serial.deserialize(this.fieldType, node);
138            try {
139                if (newVal == null) {
140                    Object existingVal = field.get(instance);
141                    if (existingVal != null) {
142                        serializeTo(instance, node);
143                    }
144                } else {
145                    field.set(instance, newVal);
146                }
147            } catch (IllegalAccessException e) {
148                throw new ObjectMappingException("Unable to deserialize field " + field.getName(), e);
149            }
150        }
151
152        @SuppressWarnings({"rawtypes", "unchecked"})
153        public void serializeTo(Object instance, ScopedConfigurationNode<?> node) throws ObjectMappingException {
154            try {
155                Object fieldVal = this.field.get(instance);
156                if (fieldVal == null) {
157                    node.setValue(null);
158                } else {
159                    TypeSerializer serial = node.getOptions().getSerializers().get(this.fieldType);
160                    if (serial == null) {
161                        throw new ObjectMappingException("No TypeSerializer found for field " + field.getName() + " of type " + this.fieldType);
162                    }
163                    serial.serialize(this.fieldType, fieldVal, node);
164                }
165
166                if (node instanceof CommentedConfigurationNodeIntermediary<?> && this.comment != null && !this.comment.isEmpty()) {
167                    CommentedConfigurationNodeIntermediary<?> commentNode = ((CommentedConfigurationNodeIntermediary<?>) node);
168                    if (!commentNode.getComment().isPresent()) {
169                        commentNode.setComment(this.comment);
170                    }
171                }
172            } catch (IllegalAccessException e) {
173                throw new ObjectMappingException("Unable to serialize field " + field.getName(), e);
174            }
175        }
176    }

178    /**
179     * Represents an object mapper bound to a certain instance of the object
180     */
181    public class BoundInstance {
182        private final T boundInstance;
183
184        protected BoundInstance(T boundInstance) {
185            this.boundInstance = boundInstance;
186        }
187
188        /**
189         * Populate the annotated fields in a pre-created object
190         *
191         * @param <Node> The type of node being populated
192         * @param source The source to get data from
193         * @return The object provided, for easier chaining
194         * @throws ObjectMappingException If an error occurs while populating data
195         */
196        public <Node extends ScopedConfigurationNode<Node>> T populate(Node source) throws ObjectMappingException {
197            for (Map.Entry<String, FieldData> ent : cachedFields.entrySet()) {
198                Node node = source.getNode(ent.getKey());
199                ent.getValue().deserializeFrom(boundInstance, node);
200            }
201            return boundInstance;
202        }
203
204        /**
205         * Serialize the data contained in annotated fields to the configuration node.
206         *
207         * @param target The target node to serialize to
208         * @param <N> The type of node being serialized to
209         * @throws ObjectMappingException if serialization was not possible due to some error.
210         */
211        public <N extends ScopedConfigurationNode<N>> void serialize(N target) throws ObjectMappingException {
212            for (Map.Entry<String, FieldData> ent : cachedFields.entrySet()) {
213                N node = target.getNode(ent.getKey());
214                ent.getValue().serializeTo(boundInstance, node);
215            }
216        }
217
218        /**
219         * Return the instance this mapper is bound to.
220         *
221         * @return The active instance
222         */
223        public T getInstance() {
224            return boundInstance;
225        }
226    }
227
228    /**
229     * Create a new object mapper of a given type. The given type must not be an interface.
230     *
231     * @param type The type this object mapper will work with
232     * @throws ObjectMappingException When errors occur discovering fields in the class
233     */
234    @SuppressWarnings("unchecked")
235    protected ObjectMapper(TypeToken<T> type) throws ObjectMappingException {
236        this.type = type;
237        this.clazz = type.getRawType();
238        if (this.clazz.isInterface()) {
239            throw new ObjectMappingException("ObjectMapper can only work with concrete types");
240        }
241
242        Invokable<T, T> constructor = null;
243        try {
244            constructor = type.constructor(type.getRawType().getDeclaredConstructor());
245            constructor.setAccessible(true);
246        } catch (NoSuchMethodException ignore) {
247        }
248        this.constructor = constructor;
249        TypeToken<? super T> collectType = type;
250        Class<? super T> collectClass = type.getRawType();
251        while (true) {
252            collectFields(cachedFields, collectType);
253            collectClass = collectClass.getSuperclass();
254            if (collectClass.equals(Object.class)) {
255                break;
256            }
257            collectType = collectType.getSupertype((Class) collectClass);
258        }
259    }
260
261    protected void collectFields(Map<String, FieldData> cachedFields, TypeToken<? super T> clazz) throws ObjectMappingException {
262        for (Field field : clazz.getRawType().getDeclaredFields()) {
263            if (field.isAnnotationPresent(Setting.class)) {
264                Setting setting = field.getAnnotation(Setting.class);
265                String path = setting.value();
266                if (path.isEmpty()) {
267                    path = field.getName();
268                }
269
270                TypeToken<?> fieldType = clazz.resolveType(field.getGenericType());
271
272                FieldData data = new FieldData(field, setting.comment(), fieldType);
273                field.setAccessible(true);
274                if (!cachedFields.containsKey(path)) {
275                    cachedFields.put(path, data);
276                }
277            }
278        }
279    }

281    /**
282     * Create a new instance of an object of the appropriate type. This method is not
283     * responsible for any population.
284     *
285     * @return The new object instance
286     * @throws ObjectMappingException If constructing a new instance was not possible
287     */
288    protected T constructObject() throws ObjectMappingException {
289        if (constructor == null) {
290            throw new ObjectMappingException("No zero-arg constructor is available for class " + type + " but is required to construct new instances!");
291        }
292        try {
293            return constructor.invoke(null);
294        } catch (IllegalAccessException | InvocationTargetException e) {
295            throw new ObjectMappingException("Unable to create instance of target class " + type, e);
296        }
297    }
298
299    /**
300     * Returns whether this object mapper can create new object instances. This may be
301     * false if the provided class has no zero-argument constructors.
302     *
303     * @return Whether new object instances can be created
304     */
305    public boolean canCreateInstances() {
306        return constructor != null;
307    }
308
309    /**
310     * Return a view on this mapper that is bound to a single object instance
311     *
312     * @param instance The instance to bind to
313     * @return A view referencing this mapper and the bound instance
314     */
315    public BoundInstance bind(T instance) {
316        return new BoundInstance(instance);
317    }
318
319    /**
320     * Returns a view on this mapper that is bound to a newly created object instance
321     *
322     * @see #bind(Object)
323     * @return Bound mapper attached to a new object instance
324     * @throws ObjectMappingException If the object could not be constructed correctly
325     */
326    public BoundInstance bindToNew() throws ObjectMappingException {
327        return new BoundInstance(constructObject());
328    }
329
330    public TypeToken<T> getMappedType() {
331        return this.type;
332    }
